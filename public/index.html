    <script>

        // --- LOGIC TO TALK TO GEMINI SERVER ---

        let threadId = null;


        document.getElementById('fileInput').addEventListener('change', function() {

            if(this.files[0]) {

                document.getElementById('fileName').innerText = this.files[0].name;

                document.getElementById('fileBadge').style.display = 'block';

                document.getElementById('sendBtn').classList.add('active');

            }

        });


        function clearFile() {

            document.getElementById('fileInput').value = '';

            document.getElementById('fileBadge').style.display = 'none';

        }


        function handleEnter(e) { if(e.key === 'Enter') sendMessage(); }


        async function sendMessage() {

            const input = document.getElementById('userInput');

            const fileInput = document.getElementById('fileInput');

            const history = document.getElementById('chatHistory');

            const text = input.value.trim();

            const file = fileInput.files[0];


            if(!text && !file) return;


            // 1. Add User Bubble

            let userHTML = `<div class="message user"><div class="avatar user">YOU</div><div class="bubble user">`;

            if(file) userHTML += `<div style="color:#d4af37; font-size:0.8rem; margin-bottom:5px;">📎 ${file.name}</div>`;

            if(text) userHTML += text;

            userHTML += `</div></div>`;

            history.innerHTML += userHTML;

            history.scrollTop = history.scrollHeight;


            // 2. Loading State

            const loadingId = 'load-' + Date.now();

            history.innerHTML += `<div id="${loadingId}" class="message"><div class="avatar ai">ZH</div><div class="bubble" style="color:#777;">Analyzing leverage...</div></div>`;

            

            // Prepare Data

            const formData = new FormData();

            formData.append('message', text);

            formData.append('threadId', threadId || "");

            if(file) formData.append('file', file);


            // Clear Input

            input.value = "";

            clearFile();


            try {

                // Send to Gemini Server

                const res = await fetch('/audit', { method: 'POST', body: formData });

                const data = await res.json();

                

                document.getElementById(loadingId).remove();

                threadId = data.threadId || threadId;


                // Format Gemini Response

                const formatted = (data.response || "Error")

                    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')

                    .replace(/\n/g, '<br>');


                history.innerHTML += `<div class="message"><div class="avatar ai">ZH</div><div class="bubble">${formatted}</div></div>`;


            } catch(e) {

                document.getElementById(loadingId).innerHTML = `<div class="bubble" style="color:red">Connection lost.</div>`;

            }

            history.scrollTop = history.scrollHeight;

        }

    </script>
